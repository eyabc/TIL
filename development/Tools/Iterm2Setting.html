  • bash-3.2# 접속
$ sudo -s

# home brew

  • 아래 주소의 내용 복사하여 터미널에서 실행 https://brew.sh/index_ko

  • Press RETURN to continue or any other key to abort 문구가 출력되면 RETURN(ENTER)키를 눌러주고 맥 password 입력

# 이름부분 Macbook-pro 지우기

$ vi .zshrc
--- insert
prompt_context() {
  if [[ "$USER" != "$DEFAULT_USER" || -n "$SSH_CLIENT" ]]; then
    prompt_segment black default "%(!.%{%F{yellow}%}.)$USER"
  fi
}
---
$ source .zshrc

# New Line

명령어가 화면에서 벗어나는 경우 new line 적용

  • agnoster theme update
$ vi .oh-my-zsh/themes/agnoster.zsh-theme
--- change
build_prompt() {
  // ...
  prompt_hg
  prompt_newline // added, sequence important
  prompt_end
}

// insert
prompt_newline() {
  if [[ -n $CURRENT_BG ]]; then
    echo -n "%{%k%F{$CURRENT_BG}%}$SEGMENT_SEPARATOR
%{%k%F{blue}%}$SEGMENT_SEPARATOR"
  else
    echo -n "%{%k%}"
  fi

  echo -n "%{%f%}"
  CURRENT_BG=''
}

---
